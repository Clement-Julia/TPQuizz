CREATE TABLE utilisateurs (
    idUtilisateur INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    email VARCHAR(100) UNIQUE NOT NULL,
    mdp VARCHAR(100) NOT NULL,
    pseudo VARCHAR(100) NOT NULL,
    reponse_secrete VARCHAR(255) NOT NULL,
	idQuestionS INT NOT NULL,
    idRole INT NOT NULL
    )

CREATE TABLE role (
    idRole INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    libelle VARCHAR(100) NOT NULL
    )

CREATE TABLE question_secrete (
    idQuestionS INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    quesion VARCHAR(100) NOT NULL
    )

CREATE TABLE quizz (
    idQuizz INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    titre VARCHAR(100) NOT NULL,
    validationAdmin BOOLEAN NOT NULL,
    idCategorie INT NOT NULL,
    idUtilisateur INT NOT NULL
    )

CREATE TABLE categories (
    idCategorie INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    libelle VARCHAR(100) NOT NULL
    )

CREATE TABLE questions (
    idQuestion INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    description VARCHAR(100) NOT NULL,
    reponse INT NOT NULL,
    idQuizz INT NOT NULL
    )

CREATE TABLE score (
    idUtilisateur INT NOT NULL,
    idQuizz INT NOT NULL,
    score INT NOT NULL,
    PRIMARY KEY (idUtilisateur, idQuizz)
    )

CREATE TABLE amis (
    idUtilisateur1 INT NOT NULL,
    idUtilisateur2 INT NOT NULL,
    PRIMARY KEY (idUtilisateur1, idUtilisateur2)
    )

CREATE TABLE reponses_quizz (
    idUtilisateur INT NOT NULL,
    idReponse INT NOT NULL,
    idQuestion INT NOT NULL,
    PRIMARY KEY (idUtilisateur, idReponse, idQuestion)
    )

CREATE TABLE reponses (
    idReponse INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    reponse VARCHAR(100) NOT NULL,
    vrai BOOLEAN NOT NULL,
    idQuestion INT NOT NULL
    )
